## Usage

### with gulp

```js
var plugin = plugins('gulp-*');

gulp.task('default', function () {
  gulp.src('test/*.js')
    .pipe(plugin.jshint());
    .pipe(plugin.mocha());
});
```

### with assemble

```js
var plugin = plugins('assemble-*');

assemble.task('default', function () {
  assemble.src('test/*.js')
    .pipe(plugin.jshint());
    .pipe(plugin.mocha());
});
```
